# Write your own code

list_datasets = ["data_quiz/wdi_ages1980.csv", "data_quiz/wdi_ages2000.csv", "data_quiz/wdi_ages2020.csv"]
list_years = [1980, 2000, 2020]

index = 0
for data in list_datasets:
    dataset = pd.read_csv(data)
    percent_ages0to14 = dataset["percent_ages0to14"]
    percent_ages15to64 = dataset["percent_ages15to64"]
    percent_ages0to64 = percent_ages0to14 + percent_ages15to64
    plt.hist(percent_ages0to64)
    plt.title(f"The Distribution of Countries with the Percentage of \nthe Population Aged Betwee 0 and 64, in Year {list_years[index]}")
    plt.xlabel("Percentage of Population Aged Between 0 and 64")
    plt.ylabel("Frequency")
    plt.show()
    index += 1
